function uuid() {
  return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, c => {
    const r = Math.random() * 16 | 0;
    const v = c === 'x' ? r : (r & 0x3 | 0x8);
    return v.toString(16);
  });
}

const points = items.map(item => {
  const embedding = item.json.embedding || item.json.data?.[0]?.embedding;

  if (!embedding || embedding.length !== 1536) {
    throw new Error("Invalid embedding detected");
  }

  return {
    id: uuid(),
    vector: embedding,
    payload: {
      text: item.json.text
    }
  };
});

return [{ json: { points } }];

const items = $input.first().json.result;
const query = $('get Question').first().json.question;
const topk = 3;
const threshold = 0.7;
// Step 1: clean + sort
const cleaned = items.map(item => {
  const text = (item.payload?.text || '')
    .replace(/[^\x00-\x7F]/g, '')
    .replace(/\s+/g, ' ')
    .trim();

  return {
    id: item.id,
    score: item.score,
    text
  };
}).sort((a, b) => b.score - a.score);
